\begin{frame}{Metodologia}
  \framesubtitle{Estratégias de recomendação}
  \begin{itemize}[<+->]
    \item Baseada em conteúdo
    \begin{itemize}
      \item Técnicas: \textit{BM25} e \textit{Bayes} ingênuo
      \item Dados: \textit{tags} e descrições dos pacotes
    \end{itemize}
    \item Colaborativa
    \begin{itemize}
      \item Técnica: \textit{K-NN}
      \item Dados: listas de pacotes instalados
    \end{itemize}
    \item Híbrida
    \begin{itemize}
      \item Cascata: refinamento de resultados a partir de \\
            \hspace{1.5cm} dados demográficos e reputação dos itens
      \item Meta-nível: colaboração por conteúdo
      \item Revezamento: decisão a partir da seleção de atributos
      \item Combinação dos resultados
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Metodologia}
  \framesubtitle{Codificação}
  \begin{itemize}[<+->]
    \item Linguagem \textit{Python}
    \item Bibliotecas: \textit{Xapian} e \textit{NLTK}
    \item Documentação: \textit{Doxygen}
    \item Controle de versão: \textit{Git}
  \end{itemize}
\end{frame}

\begin{frame}{Metodologia}
  \framesubtitle{Avaliação}
  \begin{itemize}[<+->]
    \item Testes preliminares: validação cruzada
    \item Seleção de estratégias e ajustes das soluções
    \item Consulta pública: \textit{survey} eletrônico
    \begin{enumerate}
      \item Composição da identidade do usuário
      \item Produção das recomendações por diversas estratégias
      \item Apresentação dos resultados
      \item Avaliação pelo usuário
      \item Aplicação de métricas
    \end{enumerate}
    \item Compilação dos resultados globais
  \end{itemize}
\end{frame}
